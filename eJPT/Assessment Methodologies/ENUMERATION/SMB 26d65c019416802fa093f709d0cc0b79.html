#!/bin/bash

# ---------------------------------------------
# SMB Share Enumeration Script (Anonymous Access)
# ---------------------------------------------
# This script checks which SMB shares on a target
# are accessible without authentication (guest access).
# It uses smbclient to test each share from a wordlist.
# ---------------------------------------------

# üîß Configuration
TARGET="target.ine.local"    # Target hostname or IP
WORDLIST="/root/Desktop/wordlists/shares.txt"  # Path to wordlist file

# ‚úÖ Check if wordlist file exists
if [ ! -f "$WORDLIST" ]; then
    echo "‚ùå Wordlist not found: $WORDLIST"
    exit 1
fi

echo "üîç Starting SMB share enumeration on: $TARGET"
echo "üìñ Using wordlist: $WORDLIST"
echo

# üîÅ Loop through each share in the wordlist
while read -r SHARE; do
    # Skip empty lines or comments
    [[ -z "$SHARE" || "$SHARE" =~ ^# ]] && continue

    echo "‚û°Ô∏è  Testing share: $SHARE"

    # Attempt anonymous access using smbclient
    smbclient "//$TARGET/$SHARE" -N -c "ls" &>/dev/null

    if [ $? -eq 0 ]; then
        echo "‚úÖ [OPEN] Anonymous access allowed for: $SHARE"
    else
        echo "‚õî [DENIED] Access denied for: $SHARE"
    fi

    echo
done < "$WORDLIST"

echo "‚úÖ Enumeration completed."

SMBCLIENT SYNTAX

Basic CLI syntax

smbclient //<server>/<share> [options]

Examples:

  • smbclient //10.10.10.5/public -U alice
  • smbclient //10.10.10.5 -L (list shares)

Useful options

  • U <user> ‚Äî username (can include password: U 'user%pass')
  • N ‚Äî no password (anonymous)
  • L //<server> ‚Äî list all shares on server
  • I <ip> ‚Äî connect to specific IP (skip DNS)
  • p <port> ‚Äî port (139 or 445)
  • W <domain> ‚Äî domain/workgroup
  • c "<commands>" ‚Äî run smbclient commands non-interactively
  • m <SMB1|SMB2|SMB3> ‚Äî force SMB protocol version

Examples

1) List shares (anonymous)

smbclient -L //10.10.10.5 -N

2) List shares with username (inline password)

smbclient -L //10.10.10.5 -U 'bob%P@ssw0rd'

3) Connect to a share interactively

smbclient //10.10.10.5/public -U alice
